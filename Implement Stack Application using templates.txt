#include <iostream>
using namespace std;

#define max_size 5

template<class Type>
class Stack {
private:
    static int TOP;
    Type stk[max_size];

public:
    Stack() {
        TOP = -1;
    }

    bool isFull() {
        return (TOP == max_size - 1);
    }

    bool isEmpty() {
        return (TOP == -1);
    }

    void push(Type element) {
        if (isFull()) {
            cout << "Stack overflow" << endl;
        } else {
            TOP++;
            stk[TOP] = element;
        }
    }

    void pop() {
        if (isEmpty()) {
            cout << "Stack underflow" << endl;
        } else {
            TOP--;
        }
    }

    void display_stack() {
        if (isEmpty()) {
            cout << "Stack is empty" << endl;
        } else {
            for (int i = TOP; i >= 0; i--) {
                cout << stk[i] << " ";
            }
            cout << endl;
        }
    }
};

template<class Type>
int Stack<Type>::TOP;
